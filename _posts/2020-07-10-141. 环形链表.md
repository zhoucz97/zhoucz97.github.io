Floyd算法的阶段一可解

```cpp
bool hasCycle(ListNode *head) {
    set<ListNode*> s;
    ListNode *p=head;
    while(p)
    {
        if(s.find(p)!=s.end())  return true;
        else    s.insert(p);
        p = p->next;
    }
    return false;
} 
```

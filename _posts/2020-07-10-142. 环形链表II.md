Floyd龟兔赛跑算法直接可解。

```cpp
ListNode *detectCycle(ListNode *head) {
    if(head == NULL || head->next == NULL)
        return NULL;
    ListNode *p = head, *q = head;
    while(p && q) {
        p = p->next;
        q = q->next;
        if(q)   q = q->next;
        if(p == q)  break;
    }
    if(!p || !q)  return NULL;
    p = head;
    while(p != q) {
        p = p->next;
        q = q->next;
    }
    return p;
}
```
